| "\t"
| "\n"
| "\r"
| "\f"
}

"class"

| "static"
| "if"
| "else"
| "while"
| "return"
| "extends"
| "new"
| "this"
| "true"
| "false"
| "null"
| "int"
| "boolean"
| "void"
| "+"
| "-"
| "*"
| "/"
| "=="
| "!="
| "<"
| ">"
| "<="
| ">="
}

   
/***********************/
/* Comment handling.   */
/***********************/
 
<DEFAULT> MORE : {
"//" : IN_SINGLE_LINE_COMMENT
| <"/**" ~["/"]> : IN_FORMAL_COMMENT
| "/*" : IN_MULTI_LINE_COMMENT
}

   
<IN_SINGLE_LINE_COMMENT> SPECIAL : {
<SINGLE_LINE_COMMENT: "\n" | "\r" | "\r\n"> : DEFAULT
}

   
<IN_FORMAL_COMMENT> SPECIAL : {
<FORMAL_COMMENT: "*/"> : DEFAULT
}

   
<IN_MULTI_LINE_COMMENT> SPECIAL : {
<MULTI_LINE_COMMENT: "*/"> : DEFAULT
}

   
<IN_SINGLE_LINE_COMMENT,IN_FORMAL_COMMENT,IN_MULTI_LINE_COMMENT> MORE : {
<~[]>
}

   
/************/
/* Literals */
/************/
 
<DEFAULT> TOKEN : {
<DECIMAL: ["1"-"9"] (["0"-"9"])* | "0">
| <OCTAL: "0" (["0"-"7"])+> : DEFAULT
| <HEXADECIMAL: "0" ["x","X"] (["0"-"9","a"-"f","A"-"F"])+>
| <BINARY: "0" ["b","B"] (["0","1"])+>
}

   
/***************/
/* Identifiers */
/***************/
 
<DEFAULT> TOKEN : {
<ID: <LETTER> (<LETTER> | <DIGIT> | "_")*> : DEFAULT
| <LETTER: ["a"-"z","A"-"Z"]> : DEFAULT
| <DIGIT: ["0"-"9"]> : DEFAULT
}

   
<DEFAULT> SKIP : {
<~[]> : {
}

NON-TERMINALS

/********************/
/* MiniJava Grammar */
/********************/

// Goal 	::= 	MainClass ( ClassDeclaration | ThreadDeclaration )*
Goal ::= MainClassDeclaration ( ClassDeclaration | ThreadDeclaration )* <EOF> 
// MainClass 	::= 	"class" Identifier "{" "public" "static" "void" "main" "(" "String" "[" "]" Identifier ")" "{" Statement "}" "}"
MainClassDeclaration ::= "class" <ID> "{" "public" "static" "void" "main" "(" "String" "[" "]" <ID> ")" "{" ( VarDeclaration )* ( Statement )* "}" "}" 
// ClassDeclaration 	::= 	"class" Identifier ( "extends" Identifier )? "{" ( VarDeclaration )* ( MethodDeclaration )* "}"
ClassDeclaration ::= "class" <ID> ( "extends" <ID> )? "{" ( VarDeclaration )* ( MethodDeclaration )* "}" 
// ThreadDeclaration 	::= 	"class" Identifier "extends" "Thread" "{" ( VarDeclaration )* ( MethodDeclaration | RunDeclaration )* "}"
ThreadDeclaration ::= "class" <ID> "extends" "Thread" "{" ( FieldDeclaration | VoidDeclaration | MethodDeclaration )* "}" 
// VarDeclaration 	::= 	Type Identifier ( = Expression )? ";"
VarDeclaration ::= Type <ID> ( "=" Expression )? ";" 
// MethodDeclaration 	::= 	"public" ( "synchronized" )? Type Identifier "(" ( Type Identifier ( "," Type Identifier )* )? ")" "{" ( VarDeclaration )* ( Statement )* "return" Expression ";" "}"
MethodDeclaration ::= "public" ( "synchronized" )? Type <ID> "(" ( Type <ID> ( "," Type <ID> )* )? ")" "{" ( VarDeclaration )* ( Statement )* "return" Expression ";" "}" 
// VoidDeclaration 	::= 	"public" "void" Identifier "(" ( Type Identifier ( "," Type Identifier )* )? ")" "{" ( VarDeclaration )* ( Statement )* "}"
VoidDeclaration ::= "public" "void" <ID> "(" ( Type <ID> ( "," Type <ID> )* )? ")" "{" ( VarDeclaration )* ( Statement )* "}" 
// FieldDeclaration 	::= 	Type Identifier ";"
FieldDeclaration ::= Type <ID> ( "=" Expression )? ";"
Statement ::= Block
| IfStatement
| WhileStatement
| ExpressionStatement
| VarDeclaration
IfStatement ::= "if" "(" Expression ")" Statement ( "else" Statement )?
Type ::= "int"
| "boolean"
| "void"
| <ID>
PrimaryExpression ::= Literal
| <ID>
| "(" Expression ")"
Expression ::= PrimaryExpression ( BinaryOperator PrimaryExpression )*
Literal ::= <DECIMAL>
| "true"
| "false"
Block ::= "{" ( Statement )* "}"
WhileStatement ::= "while" "(" Expression ")" Statement
ExpressionStatement ::= Expression ";"
BinaryOperation ::= Expression BinaryOperator Expression
BinaryOperator ::= <"+" | "-" | "*" | "/" | "==" | "!=" | "<" | ">" | "<=" | ">=">
